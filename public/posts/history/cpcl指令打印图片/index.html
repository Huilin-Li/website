    private String toHexString(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for(Byte b : bytes) {
            String hex = Integer.toHexString(b.intValue()).toUpperCase();
            if(hex.length() == 1) {
                hex = "0" + hex;
            } else if(hex.length() > 2) {
                hex = hex.substring(hex.length() - 2);
            }
            sb.append(hex);
        }
        return sb.toString();
    }
